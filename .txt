// 🔥 CONFIG FIREBASE (tienes que poner el tuyo)

  apiKey: "AIzaSyA2SgyB_hF3rSGIpVMiNdKk_P-1E6xxGyM",
  authDomain: "votacion-server.firebaseapp.com",
  databaseURL: ""https://votacion-server-default-rtdb.firebaseio.com",
  projectId: "votacion-server"
};

firebase.initializeApp(firebaseConfig);
const db = firebase.database();

let nombreUsuario = "";

// entrar con nombre (anti bots básico)
function entrar(){
let nombre = document.getElementById("nombre").value;

if(nombre.length < 2){
alert("Nombre muy corto");
return;
}

nombreUsuario = nombre;
document.getElementById("user").innerText = nombre;

document.getElementById("panel").style.display="block";
}

// votos en tiempo real
db.ref("votos").on("value",(snap)=>{
let data = snap.val() || {si:0,no:0};
document.getElementById("si").innerText = data.si;
document.getElementById("no").innerText = data.no;
});

// votar
function votar(tipo){

db.ref("votos/" + tipo).transaction(v=>{
return (v||0)+1;
});

// video dentro de la página
if(tipo === "si"){
document.getElementById("video").innerHTML=`
<h3>😢 ${nombreUsuario} votó SÍ</h3>
<video autoplay controls>
<source src="triste.mp4" type="video/mp4">
</video>`;
}else{
document.getElementById("video").innerHTML=`
<h3>🎉 ${nombreUsuario} votó NO</h3>
<video autoplay controls>
<source src="feliz.mp4" type="video/mp4">
</video>`;
}
}

// admin reset (simple)
function borrar(){
db.ref("votos").set({si:0,no:0});
alert("Votos borrados");
}
